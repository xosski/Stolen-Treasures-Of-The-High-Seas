# GhostSnare.ps1 - Red Hat Operator Detection
# Detects ghost-style ticket redemptions via pattern anomalies

$logPath = "C:\Logs\RedemptionLog.log"  # <- Set to your real log path
$ghostThresholdSeconds = 5              # Suspicious if Validate→Redeem < 5 seconds
$unitAbuseThreshold = 5                 # Too many suspicious redeems per UnitID
$ghostBarcodes = @{}
$unitSuspicion = @{}
$ghostHits = @()

function Parse-TicketLog {
    param ($lines)

    foreach ($line in $lines) {
        if ($line -match 'ValidateTicket.*Barcode":"(?<barcode>\d+)"') {
            $barcode = $matches['barcode']
            $validateTime = [datetime]$line.Substring(0, 32)
            $ghostBarcodes[$barcode] = @{
                ValidateTime = $validateTime
                UnitID = ''
                Redeemed = $false
                PlayerName = ''
                RedeemTime = $null
            }
        }

        if ($line -match 'Request validate ticket.*<Message.*UnitID="(?<unit>.*?)".*?<Barcode>(?<barcode>\d+)</Barcode>') {
            if ($ghostBarcodes[$matches['barcode']]) {
                $ghostBarcodes[$matches['barcode']].UnitID = $matches['unit']
            }
        }

        if ($line -match 'Received redeem ticket.*<Barcode>(?<barcode>\d+)</Barcode>.*<PlayerName>(?<name>.*?)</PlayerName>') {
            $barcode = $matches['barcode']
            $name = $matches['name']
            $redeemTime = [datetime]$line.Substring(0, 32)

            if ($ghostBarcodes[$barcode]) {
                $ghostBarcodes[$barcode].Redeemed = $true
                $ghostBarcodes[$barcode].RedeemTime = $redeemTime
                $ghostBarcodes[$barcode].PlayerName = $name
            }
        }

        # Jackpot service errors
        if ($line -match 'KonamiJackpotService.*actively refused') {
            Write-Warning "⚠️ Jackpot Service Offline Detected @ $($line.Substring(0, 32))"
        }
    }
}

function Analyze-Anomalies {
    foreach ($barcode in $ghostBarcodes.Keys) {
        $entry = $ghostBarcodes[$barcode]
        if (-not $entry.Redeemed) { continue }

        $delta = ($entry.RedeemTime - $entry.ValidateTime).TotalSeconds
        $isGhost = ($entry.PlayerName -eq "Unknown, Unknown Unknown") -and ($delta -le $ghostThresholdSeconds)

        if ($isGhost) {
            $ghostHits += $entry
            Write-Host "`n👻 Suspected Ghost Redemption:"
            Write-Host "  ▪ Barcode: $barcode"
            Write-Host "  ▪ UnitID : $($entry.UnitID)"
            Write-Host "  ▪ Time   : $($entry.ValidateTime.ToShortTimeString()) → $($entry.RedeemTime.ToShortTimeString()) ($delta sec)"
            Write-Host "  ▪ Player : $($entry.PlayerName)`n"

            # Track abuse per kiosk
            $unitSuspicion[$entry.UnitID] += 1
        }
    }

    # Detect abused kiosks
    foreach ($unit in $unitSuspicion.Keys) {
        if ($unitSuspicion[$unit] -gt $unitAbuseThreshold) {
            Write-Host "🔥 ALERT: UnitID '$unit' has $($unitSuspicion[$unit]) ghost redemptions! LOCKDOWN ADVISED!"
        }
    }
}

# Run it
$logLines = Get-Content $logPath -Tail 2000
Parse-TicketLog -lines $logLines
Analyze-Anomalies
